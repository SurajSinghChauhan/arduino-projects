    }

          digitalWrite(led, HIGH);

         }

Burglar Alarm using IR Sensor

#define led 4      // led at pin 4
#define buzzer 5      // buzzer at pin 5
#define sensor 6      // ir sensor at pin 6
int sound=250;      // set buzzer sound
void setup() {
     Serial.begin(9600);
     pinMode(sensor,INPUT);
     pinMode(led,OUTPUT);
     pinMode(buzzer,OUTPUT);
     }

void loop(){
     int detect=digitalRead(sensor);      // read status of sensor
     if(detect==HIGH) {     // if sensor detects obstacle
          digitalWrite(led,HIGH);      // led on
          tone(buzzer,sound);      // buzzer sounds
         }
     else{
          digitalWrite(led,LOW);
          noTone(buzzer);
         }
     delay(300);
     }


Temperature Reflector

float temp = 0;
void setup() {
     pinMode(A0, INPUT);
     pinMode(13, OUTPUT);
     pinMode(12, OUTPUT);
     pinMode(11, OUTPUT);
     pinMode(10, OUTPUT);
    }
void loop(){
    temp = analogRead(A0);
    temp = ((temp*5/1023)-0.5)*100;
    if (temp < 10) {
          noTone(13);
          digitalWrite(11, HIGH);
          digitalWrite(10, LOW);
          digitalWrite(12, LOW);
    }
    else if (temp <50 && temp>=10) {
          tone(13, 95);
          digitalWrite(10, HIGH);
          digitalWrite(11, LOW);
          digitalWrite(12, LOW);
    }
    else {
          tone(13, 225);
          digitalWrite(12, HIGH);
          digitalWrite(10, LOW);
          digitalWrite(11, LOW);
     }
     delay(10);      // Delay a little bit to improve simulation performance
    }

Speed control of D.C. Motor

void loop() {
     directionControl();
     delay(1000);
     speedControl();
     delay(1000);
     }
void directionControl() {      // Set motors to maximum speed
     analogWrite(9, 255);
     analogWrite(3, 255);

     // Turn on motor A & B
     digitalWrite(8, HIGH);
     digitalWrite(7, LOW);
     digitalWrite(5, HIGH);
     digitalWrite(4, LOW);
     delay(2000);

     // Now change motor directions
     digitalWrite(8, LOW);
     digitalWrite(7, HIGH);
     digitalWrite(5, LOW);
     digitalWrite(4, HIGH);
     delay(2000);
    
     // Turn off motors
     digitalWrite(8, LOW);
     digitalWrite(7, LOW);
     digitalWrite(5, LOW);
     digitalWrite(4, LOW);
    }


Temperature Display on 16×2 LCD Module

#include<LiquidCrystal.h>
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
const int sensor=A1;      // Assigning analog pin A1 to variable 'sensor'
float tempc;      //variable to store temperature in degree Celsius
float tempf;      //variable to store temperature in Fahreinheit
float vout;      //temporary variable to hold sensor reading
void setup(){
     pinMode(sensor,INPUT);      // Configuring pin A1 as input
     Serial.begin(9600);
     lcd.begin(16,2);
     delay(500);
    }

void loop(){
     vout=analogRead(sensor);
     vout=((vout*5/1024)-0.5)*100;
     tempc=vout;      // Storing value in Degree Celsius
     tempf=(vout*1.8)+32;      // Converting to Fahrenheit
     lcd.setCursor(0,0);
     lcd.print("in DegreeC= ");
     lcd.print(tempc);
     lcd.setCursor(0,1);
     lcd.print("in Fahrenheit=");
     lcd.print(tempf);
     delay(1000);      //Delay of 1 second for ease of viewing in serial monitor
    }
